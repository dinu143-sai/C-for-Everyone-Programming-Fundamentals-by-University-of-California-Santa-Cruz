#include <stdio.h>
#define PI 3.14159

int main(void) {
    int radius;
    printf("Enter radius: ");
    scanf("%d", &radius);
    
    // Correct formula and use of double for floating-point calculation
    double volume = (4.0 / 3.0) * PI * radius * radius * radius;
    
    printf("Volume is: %.2f \n\n", volume);
    
    return 0;
}
